// Funksjon for å endre et input-felt til en dropdown-liste, eller omvendt
function ChangeInputToDropdownOrDropdownToInput() {
    // Logger IDen til "changeToDropdown"-knappen
    debugLog('changeToDropdown-knappen:', $('#changeToDropdown')[0]);

    // Legger til en event listener for "changeToDropdown"- og "changeToInput"-knappene
    $('#changeToDropdown, #changeToInput').click(function () {
        // Sjekker om IDen til den klikkede knappen er "changeToInput"
        if (this.id === 'changeToInput') {

            // Deaktiverer "button-skannjobb"-knappen og endrer dens klasse til "btn btn-secondary"
            $('#button-skannjobb').attr('disabled', true);
            $('#button-skannjobb').removeClass().addClass('btn btn-secondary');

            // Henter dropdown-listen og erstatter den med et input-felt
            var dropdown = document.getElementById('filnavn');
            var inputField = document.createElement('input');
            inputField.id = 'filnavn';
            inputField.name = 'filnavn'; // Legger til et 'name'-attributt til input-feltet
            inputField.type = 'text';
            inputField.classList.add('form-control'); // Legger til klassen 'form-control' til input-feltet
            dropdown.parentNode.replaceChild(inputField, dropdown);

            // Endrer IDen og klassen til den klikkede knappen
            this.id = 'changeToDropdown';
            this.className = 'btn btn-primary'; // Endrer klassen tilbake til 'btn btn-primary'
            $('#eksisterende_til_sletting').val(0);
        } else {
            // Logger at "changeToDropdown"-knappen ble klikket
            debugLog('changeToDropdown-knappen ble klikket');

            // Aktiverer "button-skannjobb"-knappen og endrer dens klasse til "btn btn-primary"
            $('#button-skannjobb').attr('disabled', false);
            $('#button-skannjobb').removeClass().addClass('btn btn-primary');

            // Henter input-feltet og erstatter det med en dropdown-liste
            var inputField = document.getElementById('filnavn');
            var dropdown = document.createElement('select');
            dropdown.id = 'filnavn'; // Gir den nye dropdown-listen samme ID som det gamle input-feltet
            dropdown.name = 'filnavn'; // Legger til et 'name'-attributt til dropdown-listen
            dropdown.classList.add('form-select'); // Legger til klassen 'form-select' til dropdown-listen
            inputField.parentNode.replaceChild(dropdown, inputField);

            // Endrer IDen og klassen til den klikkede knappen
            var button = document.getElementById('changeToDropdown');
            button.id = 'changeToInput'; // Endrer IDen til 'changeToInput'
            button.className = 'btn btn-secondary';

            // Henter data fra den eksterne JSON-filen
            var url = ureg_bok;

            // Legger til tidsstempel til URL for å unngå caching
            url = url + "?_=" + new Date().getTime();

fetch(url + '?t=' + new Date().getTime())
    .then(response => response.json())
    .then(data => {

        // Itererer gjennom hvert objekt i JSON-dataene
        data.forEach(item => {
            // Oppretter en ny <option> for hvert objekt
            var option = document.createElement('option');
            option.value = item.filnavn;
            option.text = item.filnavn;
            dropdown.add(option);
        });
    })
    .catch(error => console.error('Error:', error));
            $('#eksisterende_til_sletting').val(1);
        }
    });
}

// Funksjon for å håndtere klikk på "button-sjekk-isbn"-knappen
function isbnButtonEventListener() {
    // Logger IDen til "button-sjekk-isbn"-knappen
    debugLog('button-sjekk-isbn-knappen:', $('#button-sjekk-isbn')[0]);

    // Legger til en event listener for "button-sjekk-isbn"-knappen
    $('#button-sjekk-isbn').click(function () {
        // Logger at "button-sjekk-isbn"-knappen ble klikket
        debugLog('button-sjekk-isbn-knappen ble klikket');

        // Henter ISBN-nummeret fra tekstfeltet
        var isbnNumber = $('#isbn').val();

        // Henter data fra JSON-filen
        var url = json_bok_unik_isbn;

        // Legger til tidsstempel til URL for å unngå caching
        url = url + "?_=" + new Date().getTime();

        fetch(url)
            .then(response => response.json())
            .then(data => {
                // Finner objektet som inneholder det matchende ISBN-nummeret
                var matchingItem = data.response.data.find(item => item.fieldData.isbn === isbnNumber);
                if (matchingItem) {
                    // Logger at ISBN-nummeret finnes i JSON-filen
                    debugLog('Nummeret finnes i JSON-filen');

                    // Fyller ut feltene med data fra objektet
                    $('#boktittel').val(matchingItem.fieldData.boktittel).attr('readonly', true).attr('disabled', true);
                    $('#forfattere').val(matchingItem.fieldData.forfatter).attr('readonly', true).attr('disabled', true);
                    $('#utgitt').val(matchingItem.fieldData.utgitt).attr('readonly', true).attr('disabled', true);
                    $('#forlag').val(matchingItem.fieldData.forlag).attr('readonly', true).attr('disabled', true);
                    $('#utgave').val(matchingItem.fieldData.merknad).attr('readonly', true).attr('disabled', true);

                    // Setter found_isbn til true
                    found_isbn = true;

                    // Oppdaterer lastIsbn
                    lastIsbn = isbnNumber;

                } else {
                    // Logger at ISBN-nummeret ikke finnes i JSON-filen
                    debugLog('Nummeret finnes ikke i JSON-filen');
                }
            })
            .catch(error => console.error('Error:', error));
    });
}

// Funksjon for å håndtere klikk på "button-rens-isbn"-knappen
function resetButtonEventListener() {
    // Logger IDen til "button-rens-isbn"-knappen
    debugLog('button-rens-isbn-knappen:', $('#button-rens-isbn')[0]);

    // Legger til en event listener for "button-rens-isbn"-knappen
    $('#button-rens-isbn').click(function () {
        // Logger at "button-rens-isbn"-knappen ble klikket
        debugLog('button-rens-isbn-knappen ble klikket');

        // Henter ISBN-nummeret fra tekstfeltet
        var isbn = $('#isbn').val();

        // Sjekker om found_isbn er true og isbn er forskjellig fra lastIsbn
        if (found_isbn === true && isbn !== lastIsbn) {
            // Tømmer feltene og setter 'readonly' og 'disabled' attributtene til false
            $('#boktittel').val('').attr('readonly', false).attr('disabled', false);
            $('#forfattere').val('').attr('readonly', false).attr('disabled', false);
            $('#utgitt').val('').attr('readonly', false).attr('disabled', false);
            $('#forlag').val('').attr('readonly', false).attr('disabled', false);

            // Oppdaterer lastIsbn
            lastIsbn = isbn;
        }
    });
}

// Funksjon for å håndtere klikk på "button-skannjobb"-knappen
function scanJobButtonEventListener() {
    // Logger IDen til "button-skannjobb"-knappen
    debugLog('button-skannjobb-knappen:', $('#button-skannjobb')[0]);

    // Legger til en event listener for "button-skannjobb"-knappen
    $('#button-skannjobb').click(function () {
        // Logger at "button-skannjobb"-knappen ble klikket
        debugLog('button-skannjobb-knappen ble klikket');

        // Henter filnavnet fra tekstfeltet
        var fileName = $('#filnavn').val();

        // Henter data fra JSON-filen
        var url = ureg_bok; // Erstatt med stien til JSON-filen din

        // Legger til tidsstempel til URL for å unngå caching
        url = url + "?_=" + new Date().getTime();

        fetch(url)
            .then(response => response.json())
            .then(data => {
                // Logger data
                debugLog(data);

                // Finner objektet som inneholder det matchende filnavnet
                var matchingItem = data.find(item => item.filnavn === fileName);
                if (matchingItem) {
                    // Logger at filnavnet finnes i JSON-filen
                    debugLog('Filnavnet finnes i JSON-filen');

                    // Setter verdien til "skannjobb_id"-feltet til "skannjobb_id"-verdien fra objektet
                    $('#skannjobb_id').val(matchingItem.skannjobb_id);

                    // Endrer klassen til knappen til "btn btn-secondary"
                    // Deaktiverer knappen for å hente skannjobb_id
                    $('#button-skannjobb').attr('disabled', true);
                    $('#button-skannjobb').removeClass().addClass('btn btn-secondary');

                } else {
                    // Logger at filnavnet ikke finnes i JSON-filen
                    debugLog('Filnavnet finnes ikke i JSON-filen');
                }
            })
            .catch(error => console.error('Error:', error));
    });
}

// Funksjon for å håndtere klikk på "rens_data"-knappen
function clearDataButtonEventListener() {
    // Logger IDen til "rens_data"-knappen
    debugLog('rens_data-knappen:', $('#rens_data')[0]);

    // Legger til en event listener for "rens_data"-knappen
    $('#rens_data').click(function () {
        // Logger at "rens_data"-knappen ble klikket
        debugLog('rens_data-knappen ble klikket');

        // Tømmer feltene og setter 'readonly' og 'disabled' attributtene til false
        $('#boktittel').val('').attr('readonly', false).attr('disabled', false);
        $('#forfattere').val('').attr('readonly', false).attr('disabled', false);
        $('#utgitt').val('').attr('readonly', false).attr('disabled', false);
        $('#forlag').val('').attr('readonly', false).attr('disabled', false);
        $('#isbn').val('');
        $('#filnavn').val('');
        $('#skannjobb_id').val('');

        // Deaktiverer "button-skannjobb"-knappen og endrer dens klasse til "btn btn-secondary"
        $('#button-skannjobb').attr('disabled', true);
        $('#button-skannjobb').removeClass().addClass('btn btn-secondary');

        // Endrer IDen til knappen og oppdaterer klassene
        $('#changeToInput').attr('id', 'changeToDropdown').removeClass('btn btn-secondary').addClass('btn btn-primary');

        // Erstatter "filnavn"-dropdown med et input-felt
        $('#filnavn').replaceWith('<input type="text" class="form-control" id="filnavn" placeholder="" aria-label="Felt for filnavn" name="filnavn" value="<?php echo isset($_SESSION['form_data']['filnavn']) ? $_SESSION['form_data']['filnavn'] : ''; ?>">');
    });
}

// Funksjon som kaller alle event listener-funksjonene
function functions_bokutdragListener() {
    ChangeInputToDropdownOrDropdownToInput();
    isbnButtonEventListener();
    resetButtonEventListener();
    scanJobButtonEventListener();
    clearDataButtonEventListener();
    attachFormSubmitHandler('#bokutdragForm', bokutdrag_path);
    attachFormSubmitHandler('#isbnForm', bokutdrag_path);
} // JavaScript Document
